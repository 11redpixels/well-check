### **Purpose:**
Secure 6-digit invitation system for adding new family members without manual data entry.

### **Schema:**

```typescript
/**
 * V11.2: Sharing Code - 6-digit invitation system
 * Competitor 3: Family onboarding without manual data entry
 */
interface SharingCode {
  code: string; // 6-digit numeric code (e.g., "123456")
  tenantId: string; // Family ID
  createdBy: string; // User ID of Family Head who generated code
  createdAt: number; // Timestamp
  expiresAt: number; // Expiration timestamp (24 hours default)
  maxUses: number; // Maximum number of uses (1 = single-use, 0 = unlimited)
  usedCount: number; // Number of times code has been used
  status: 'active' | 'expired' | 'revoked';
  
  // Optional: Role assignment
  assignedRole?: UserRole; // Role to assign to new member (default: 'monitor')
  
  // Optional: Permissions
  permissions?: {
    canViewMedication: boolean;
    canViewLocation: boolean;
    canReceiveAlerts: boolean;
  };
}

/**
 * V11.2: Sharing Code Usage - Track who joined with which code
 */
interface SharingCodeUsage {
  id: string;
  codeId: string; // Reference to SharingCode
  userId: string; // User who joined
  usedAt: number; // Timestamp
  ipAddress?: string; // Optional: Security tracking
  deviceInfo?: string; // Optional: Device fingerprint
}

/**
 * V11.2: Generate Sharing Code
 * Family Head only
 */
function generateSharingCode(tenantId: string, createdBy: string, options?: {
  expiresIn?: number; // Milliseconds (default: 24 hours)
  maxUses?: number; // Default: 1 (single-use)
  assignedRole?: UserRole; // Default: 'monitor'
}): SharingCode {
  const code = Math.floor(100000 + Math.random() * 900000).toString(); // 6-digit random
  
  return {
    code,
    tenantId,
    createdBy,
    createdAt: Date.now(),
    expiresAt: Date.now() + (options?.expiresIn || 24 * 60 * 60 * 1000), // 24h default
    maxUses: options?.maxUses || 1,
    usedCount: 0,
    status: 'active',
    assignedRole: options?.assignedRole || 'monitor',
    permissions: {
      canViewMedication: true,
      canViewLocation: true,
      canReceiveAlerts: true,
    },
  };
}

/**
 * V11.2: Validate Sharing Code
 * Check if code is valid and not expired
 */
function validateSharingCode(code: string): { 
  valid: boolean; 
  error?: string; 
  sharingCode?: SharingCode 
} {
  // TODO: Look up code in database
  const sharingCode = findSharingCodeByCode(code);
  
  if (!sharingCode) {
    return { valid: false, error: 'Invalid code' };
  }
  
  if (sharingCode.status !== 'active') {
    return { valid: false, error: 'Code is expired or revoked' };
  }
  
  if (sharingCode.expiresAt < Date.now()) {
    return { valid: false, error: 'Code has expired' };
  }
  
  if (sharingCode.maxUses > 0 && sharingCode.usedCount >= sharingCode.maxUses) {
    return { valid: false, error: 'Code has reached maximum uses' };
  }
  
  return { valid: true, sharingCode };
}

/**
 * V11.2: Use Sharing Code
 * Add user to family using sharing code
 */
function useSharingCode(code: string, userId: string): { 
  success: boolean; 
  error?: string; 
  tenantId?: string;
  assignedRole?: UserRole;
} {
  const validation = validateSharingCode(code);
  
  if (!validation.valid || !validation.sharingCode) {
    return { success: false, error: validation.error };
  }
  
  const sharingCode = validation.sharingCode;
  
  // Add user to family (tenantId)
  addUserToTenant(userId, sharingCode.tenantId, sharingCode.assignedRole || 'monitor');
  
  // Record usage
  recordSharingCodeUsage({
    id: generateId(),
    codeId: sharingCode.code,
    userId,
    usedAt: Date.now(),
  });
  
  // Increment usage count
  incrementSharingCodeUsage(sharingCode.code);
  
  return { 
    success: true, 
    tenantId: sharingCode.tenantId,
    assignedRole: sharingCode.assignedRole,
  };
}
```

/**
 * V11.2: Health Vitals - Clinical health stack
 * Competitor 4: Blood Glucose, Blood Pressure, BMI, Water Intake, Heart Rate
 */
interface HealthVitals {
  userId: string;
  timestamp: number;
  
  // Heart Rate (bpm)
  heartRate?: {
    value: number; // Beats per minute
    unit: 'bpm';
    source: 'manual' | 'smartwatch' | 'blood_pressure_monitor';
  };
  
  // Blood Pressure (mmHg)
  bloodPressure?: {
    systolic: number; // Top number (e.g., 120)
    diastolic: number; // Bottom number (e.g., 80)
    unit: 'mmHg';
    source: 'manual' | 'blood_pressure_monitor';
  };
  
  // Blood Glucose (mmol/L)
  bloodGlucose?: {
    value: number; // mmol/L (e.g., 5.5)
    unit: 'mmol/L';
    source: 'manual' | 'glucose_meter';
    testType: 'fasting' | 'random' | 'post_meal'; // Type of glucose test
  };
  
  // BMI (Body Mass Index)
  bmi?: {
    value: number; // Calculated from weight/height
    weight: number; // kg
    height: number; // cm
    unit: 'kg/m²';
    source: 'manual' | 'smart_scale';
  };
  
  // Water Intake (ml)
  waterIntake?: {
    value: number; // Milliliters consumed
    unit: 'ml';
    dailyGoal: number; // Target ml per day (default: 2000ml)
    source: 'manual' | 'smart_bottle';
  };
  
  // Notes (optional)
  notes?: string; // User or caregiver notes
  
  // Location (optional - for context)
  location?: {
    lat: number;
    lng: number;
  };
}

/**
 * V11.2: Health Vitals Summary - Latest readings for member card display
 */
interface HealthVitalsSummary {
  userId: string;
  lastUpdated: number;
  
  // Latest readings
  heartRate?: { value: number; timestamp: number; trend: 'up' | 'down' | 'stable' };
  bloodPressure?: { systolic: number; diastolic: number; timestamp: number; status: 'normal' | 'elevated' | 'high' };
  bloodGlucose?: { value: number; timestamp: number; status: 'low' | 'normal' | 'high' };
  bmi?: { value: number; timestamp: number; category: 'underweight' | 'normal' | 'overweight' | 'obese' };
  waterIntake?: { value: number; dailyGoal: number; timestamp: number; percentage: number }; // Today's intake
  
  // Overall health score (0-100)
  healthScore: number;
}

/**
 * V11.2: Get Health Vitals Summary
 * Aggregate latest readings for display
 */
function getHealthVitalsSummary(userId: string): HealthVitalsSummary {
  const readings = getLatestHealthVitals(userId, 7); // Last 7 days
  
  const latest = readings[0]; // Most recent
  const previous = readings[1]; // Previous reading for trend
  
  return {
    userId,
    lastUpdated: latest?.timestamp || 0,
    
    heartRate: latest?.heartRate ? {
      value: latest.heartRate.value,
      timestamp: latest.timestamp,
      trend: previous?.heartRate 
        ? (latest.heartRate.value > previous.heartRate.value ? 'up' : 
           latest.heartRate.value < previous.heartRate.value ? 'down' : 'stable')
        : 'stable',
    } : undefined,
    
    bloodPressure: latest?.bloodPressure ? {
      systolic: latest.bloodPressure.systolic,
      diastolic: latest.bloodPressure.diastolic,
      timestamp: latest.timestamp,
      status: getBloodPressureStatus(latest.bloodPressure.systolic, latest.bloodPressure.diastolic),
    } : undefined,
    
    bloodGlucose: latest?.bloodGlucose ? {
      value: latest.bloodGlucose.value,
      timestamp: latest.timestamp,
      status: getBloodGlucoseStatus(latest.bloodGlucose.value, latest.bloodGlucose.testType),
    } : undefined,
    
    bmi: latest?.bmi ? {
      value: latest.bmi.value,
      timestamp: latest.timestamp,
      category: getBMICategory(latest.bmi.value),
    } : undefined,
    
    waterIntake: getTodaysWaterIntake(userId),
    
    healthScore: calculateHealthScore(userId, readings),
  };
}

/**
 * V11.2: Blood Pressure Status (Clinical Thresholds)
 * Source: American Heart Association guidelines
 */
function getBloodPressureStatus(systolic: number, diastolic: number): 'normal' | 'elevated' | 'high' {
  if (systolic < 120 && diastolic < 80) return 'normal';
  if (systolic < 130 && diastolic < 80) return 'elevated';
  return 'high'; // Stage 1 or Stage 2 hypertension
}

/**
 * V11.2: Blood Glucose Status (Clinical Thresholds)
 * Source: American Diabetes Association guidelines
 */
function getBloodGlucoseStatus(value: number, testType: 'fasting' | 'random' | 'post_meal'): 'low' | 'normal' | 'high' {
  if (testType === 'fasting') {
    if (value < 4.0) return 'low'; // <4.0 mmol/L = Hypoglycemia
    if (value < 5.6) return 'normal'; // 4.0-5.5 mmol/L = Normal
    return 'high'; // ≥5.6 mmol/L = Prediabetes/Diabetes
  }
  
  if (testType === 'post_meal') {
    if (value < 4.0) return 'low';
    if (value < 7.8) return 'normal'; // <7.8 mmol/L = Normal (2h post-meal)
    return 'high'; // ≥7.8 mmol/L = Prediabetes/Diabetes
  }
  
  // Random
  if (value < 4.0) return 'low';
  if (value < 11.1) return 'normal'; // <11.1 mmol/L = Normal
  return 'high'; // ≥11.1 mmol/L = Diabetes
}

/**
 * V11.2: BMI Category
 * Source: WHO guidelines
 */
function getBMICategory(bmi: number): 'underweight' | 'normal' | 'overweight' | 'obese' {
  if (bmi < 18.5) return 'underweight';
  if (bmi < 25) return 'normal';
  if (bmi < 30) return 'overweight';
  return 'obese';
}

/**
 * V11.2: Today's Water Intake
 * Aggregate all water intake entries for today
 */
function getTodaysWaterIntake(userId: string): { value: number; dailyGoal: number; timestamp: number; percentage: number } {
  const today = new Date();
  today.setHours(0, 0, 0, 0);
  
  const todayReadings = getHealthVitalsInRange(userId, today.getTime(), Date.now());
  const waterIntakes = todayReadings.filter(r => r.waterIntake).map(r => r.waterIntake!);
  
  const totalIntake = waterIntakes.reduce((sum, w) => sum + w.value, 0);
  const dailyGoal = waterIntakes[0]?.dailyGoal || 2000; // Default 2L
  
  return {
    value: totalIntake,
    dailyGoal,
    timestamp: Date.now(),
    percentage: Math.round((totalIntake / dailyGoal) * 100),
  };
}

/**

 * Track when users enter/exit geofenced zones
 */
interface GeofenceLog {

  userId: string;
  geofenceId: string; // Reference to Geofence zone
  geofenceName: string; // e.g., "Home", "School", "Grandma's House"
  eventType: 'entry' | 'exit';
  timestamp: number;
  location: { lat: number; lng: number };
  notificationSent: boolean; // Was family notified?
  notifiedUsers: string[]; // User IDs who received notification
}

/**
 * V11.2: Geofence Zone
 * Define safe zones with entry/exit monitoring
 */
interface Geofence {
  id: string;
  tenantId: string; // Family ID
  name: string; // e.g., "Home", "School", "Hospital"
  center: { lat: number; lng: number };
  radius: number; // Meters
  color: string; // Map display color

  monitorExit: boolean; // Notify on exit?
  notifyUsers: string[]; // User IDs to notify

    expectedEntry?: number; // Expected entry time (milliseconds since midnight)

  };
  
  // Status
  createdAt: number;
  createdBy: string; // User ID
  active: boolean;
}

 * V11.2: Driving Stats Summary - For member card display
 */
interface DrivingStatsSummary {
  userId: string;
  lastUpdated: number;
  
  // Last 7 days
  totalTrips: number;
  totalDistance: number; // km
  totalDuration: number; // milliseconds
  
  // Safety metrics
  topSpeed: number; // km/h (highest recorded)
  hardBrakingEvents: number;
  hardAccelerationEvents: number;
  speedingEvents: number;
  
  // Safety score (0-100)
  safetyScore: number;
  
  // Current location status
  currentLocation?: {
    lat: number;
    lng: number;
    insideGeofence?: string; // Geofence name if inside
  };
}

/**
 * V11.2: Calculate Driving Safety Score
 * 0-100 score based on driving behavior
 */
function calculateDrivingSafetyScore(stats: DrivingStats): number {
  let score = 100;
  
  // Penalize speeding (-5 points per event, max -30)
  score -= Math.min(stats.speedingEvents * 5, 30);
  
  // Penalize hard braking (-3 points per event, max -30)
  score -= Math.min(stats.hardBrakingEvents * 3, 30);
  
  // Penalize hard acceleration (-2 points per event, max -20)
  score -= Math.min(stats.hardAccelerationEvents * 2, 20);
  
  // Bonus for smooth driving (no events)
  if (stats.speedingEvents === 0 && stats.hardBrakingEvents === 0 && stats.hardAccelerationEvents === 0) {
    score += 10; // Bonus for perfect trip
  }
  
  return Math.max(0, Math.min(100, score));
}

/**
 * V11.2: Detect Hard Braking Event
 * Trigger when deceleration exceeds threshold
 */
function detectHardBraking(currentSpeed: number, previousSpeed: number, deltaTime: number): {
  isHardBraking: boolean;
  deceleration: number;
} {
  const HARD_BRAKING_THRESHOLD = 6; // m/s² (sudden stop)
  
  const deltaSpeed = previousSpeed - currentSpeed; // km/h
  const deltaSpeedMs = (deltaSpeed * 1000) / 3600; // Convert to m/s
  const deceleration = deltaSpeedMs / (deltaTime / 1000); // m/s²
  
  return {
    isHardBraking: deceleration >= HARD_BRAKING_THRESHOLD,
    deceleration,
  };
}

/**
 * V11.2: Check Geofence Entry/Exit
 * Detect when user enters/exits geofence zone
 */
function checkGeofenceEvent(
  currentLocation: { lat: number; lng: number },
  previousLocation: { lat: number; lng: number },
  geofence: Geofence
): 'entry' | 'exit' | null {
  const wasInside = isInsideGeofence(previousLocation, geofence);
  const isInside = isInsideGeofence(currentLocation, geofence);
  
  if (!wasInside && isInside) return 'entry';
  if (wasInside && !isInside) return 'exit';
  return null;
}

/**
 * V11.2: Check if location is inside geofence
 * Use Haversine formula for distance calculation
 */
function isInsideGeofence(location: { lat: number; lng: number }, geofence: Geofence): boolean {
  const distance = calculateDistance(location, geofence.center);
  return distance <= geofence.radius;
}
