```typescript
/**
 * V11.2: Blood Pressure Alert Levels
 * Immediate notification to Family Head for hypertensive crisis
 */
interface BloodPressureAlert {
  level: 'normal' | 'elevated' | 'stage1' | 'stage2' | 'crisis';
  notify: boolean; // Should Family Head be notified?
  urgency: 'none' | 'monitor' | 'immediate' | 'emergency';
  message: string;
}

function getBloodPressureAlert(systolic: number, diastolic: number): BloodPressureAlert {
  // HYPERTENSIVE CRISIS (>180/120) - EMERGENCY
  if (systolic > 180 || diastolic > 120) {
    return {
      level: 'crisis',
      notify: true,
      urgency: 'emergency',
      message: '🚨 HYPERTENSIVE CRISIS - Seek emergency medical attention immediately. BP: ' + 
               systolic + '/' + diastolic + ' mmHg (Normal: <120/80)',
    };
  }
  
  // STAGE 2 HYPERTENSION (140-179/90-119)
  if (systolic >= 140 || diastolic >= 90) {
    return {
      level: 'stage2',
      notify: true,
      urgency: 'immediate',
      message: '⚠️ HIGH BLOOD PRESSURE - Stage 2 Hypertension detected. BP: ' + 
               systolic + '/' + diastolic + ' mmHg. Contact doctor within 24 hours.',
    };
  }
  
  // STAGE 1 HYPERTENSION (130-139/80-89)
  if (systolic >= 130 || diastolic >= 80) {
    return {
      level: 'stage1',
      notify: true,
      urgency: 'monitor',
      message: '⚠️ Elevated Blood Pressure - Stage 1 Hypertension. BP: ' + 
               systolic + '/' + diastolic + ' mmHg. Monitor closely.',
    };
  }
  
  // ELEVATED (120-129/<80)
  if (systolic >= 120 && systolic < 130 && diastolic < 80) {
    return {
      level: 'elevated',
      notify: false,
      urgency: 'monitor',
      message: 'Elevated Blood Pressure - BP: ' + systolic + '/' + diastolic + ' mmHg. ' +
               'Lifestyle changes recommended.',
    };
  }
  
  // NORMAL (<120/80)
  return {
    level: 'normal',
    notify: false,
    urgency: 'none',
    message: 'Blood Pressure Normal - BP: ' + systolic + '/' + diastolic + ' mmHg',
  };
}
```

```typescript
/**
 * V11.2: Blood Glucose Alert Levels
 * Immediate notification for hypoglycemia (<3.9) or hyperglycemia (>13.9)
 */
interface BloodGlucoseAlert {
  level: 'severe_low' | 'low' | 'normal' | 'prediabetes' | 'diabetes' | 'severe_high';
  notify: boolean;
  urgency: 'none' | 'monitor' | 'immediate' | 'emergency';
  message: string;
}

function getBloodGlucoseAlert(value: number, testType: 'fasting' | 'random' | 'post_meal'): BloodGlucoseAlert {
  // SEVERE HYPOGLYCEMIA (<3.0 mmol/L) - EMERGENCY
  if (value < 3.0) {
    return {
      level: 'severe_low',
      notify: true,
      urgency: 'emergency',
      message: '🚨 SEVERE HYPOGLYCEMIA - Blood glucose critically low: ' + value + ' mmol/L. ' +
               'Consume 15g fast-acting carbs immediately. Call 911 if unconscious.',
    };
  }
  
  // HYPOGLYCEMIA (3.0-3.9 mmol/L)
  if (value < 4.0) {
    return {
      level: 'low',
      notify: true,
      urgency: 'immediate',
      message: '⚠️ LOW BLOOD GLUCOSE - Hypoglycemia detected: ' + value + ' mmol/L. ' +
               'Consume 15g carbs (juice, glucose tablets). Retest in 15 minutes.',
    };
  }
  
  // SEVERE HYPERGLYCEMIA (>13.9 mmol/L) - EMERGENCY
  if (value > 13.9) {
    return {
      level: 'severe_high',
      notify: true,
      urgency: 'emergency',
      message: '🚨 SEVERE HYPERGLYCEMIA - Blood glucose critically high: ' + value + ' mmol/L. ' +
               'Risk of diabetic ketoacidosis (DKA). Seek emergency medical attention.',
    };
  }
  
  if (testType === 'fasting') {
    // DIABETES (≥7.0 mmol/L fasting)
    if (value >= 7.0) {
      return {
        level: 'diabetes',
        notify: true,
        urgency: 'immediate',
        message: '⚠️ HIGH FASTING GLUCOSE - Diabetic range: ' + value + ' mmol/L. ' +
                 'Contact doctor to discuss treatment plan.',
      };
    }
    
    // PREDIABETES (5.6-6.9 mmol/L fasting)
    if (value >= 5.6) {
      return {
        level: 'prediabetes',
        notify: true,
        urgency: 'monitor',
        message: '⚠️ Elevated Fasting Glucose - Prediabetic range: ' + value + ' mmol/L. ' +
                 'Lifestyle changes recommended.',
      };
    }
  }
  
  if (testType === 'post_meal') {
    // DIABETES (≥11.1 mmol/L post-meal)
    if (value >= 11.1) {
      return {
        level: 'diabetes',
        notify: true,
        urgency: 'immediate',
        message: '⚠️ HIGH POST-MEAL GLUCOSE - Diabetic range: ' + value + ' mmol/L (2h post-meal). ' +
                 'Contact doctor to discuss treatment plan.',
      };
    }
    
    // PREDIABETES (7.8-11.0 mmol/L post-meal)
    if (value >= 7.8) {
      return {
        level: 'prediabetes',
        notify: true,
        urgency: 'monitor',
        message: '⚠️ Elevated Post-Meal Glucose - Prediabetic range: ' + value + ' mmol/L. ' +
                 'Consider dietary adjustments.',
      };
    }
  }
  
  // NORMAL
  return {
    level: 'normal',
    notify: false,
    urgency: 'none',
    message: 'Blood Glucose Normal - ' + value + ' mmol/L',
  };
}
```

```typescript
/**
 * V11.2: Heart Rate Alert Levels
 * Immediate notification for bradycardia (<50) or tachycardia (>100 resting)
 */
interface HeartRateAlert {
  level: 'bradycardia' | 'normal' | 'elevated' | 'tachycardia' | 'severe_tachycardia';
  notify: boolean;
  urgency: 'none' | 'monitor' | 'immediate' | 'emergency';
  message: string;
}

function getHeartRateAlert(value: number, isResting: boolean = true): HeartRateAlert {
  if (!isResting) {
    // During exercise, heart rate can be elevated (50-85% of max HR)
    // Not alerting for exercise-related increases
    return {
      level: 'normal',
      notify: false,
      urgency: 'none',
      message: 'Heart Rate (Exercise): ' + value + ' bpm',
    };
  }
  
  // SEVERE TACHYCARDIA (>120 resting) - EMERGENCY
  if (value > 120) {
    return {
      level: 'severe_tachycardia',
      notify: true,
      urgency: 'emergency',
      message: '🚨 SEVERE TACHYCARDIA - Resting heart rate critically high: ' + value + ' bpm. ' +
               'Seek emergency medical attention immediately.',
    };
  }
  
  // TACHYCARDIA (>100 resting)
  if (value > 100) {
    return {
      level: 'tachycardia',
      notify: true,
      urgency: 'immediate',
      message: '⚠️ HIGH HEART RATE - Tachycardia detected: ' + value + ' bpm (Normal: 60-100). ' +
               'Monitor closely. Contact doctor if persistent.',
    };
  }
  
  // ELEVATED (90-100)
  if (value >= 90) {
    return {
      level: 'elevated',
      notify: false,
      urgency: 'monitor',
      message: 'Elevated Heart Rate: ' + value + ' bpm. Monitor if persistent.',
    };
  }
  
  // BRADYCARDIA (<50) - For elderly or non-athletes
  if (value < 50) {
    return {
      level: 'bradycardia',
      notify: true,
      urgency: 'immediate',
      message: '⚠️ LOW HEART RATE - Bradycardia detected: ' + value + ' bpm (Normal: 60-100). ' +
               'Contact doctor if symptomatic (dizziness, fatigue).',
    };
  }
  
  // NORMAL (50-90)
  return {
    level: 'normal',
    notify: false,
    urgency: 'none',
    message: 'Heart Rate Normal: ' + value + ' bpm',
  };
}
```

```typescript
interface EnvironmentalPattern {

  patternType: 'humidity_sleep' | 'temperature_activity' | 'weather_pain' | 'pollen_asthma';
  
  // Environmental trigger
  trigger: {

    operator: '>' | '<' | '==';
  };
  
  // Health impact
  impact: {

    effect: 'worsens' | 'improves';
    correlation: number; // 0-1 (1 = perfect correlation)
  };
  
  // Evidence
  occurrences: number; // Number of times pattern observed
  confidence: number; // 0-100% confidence in pattern
  
  // Recommendation
  recommendation: string; // e.g., "Use dehumidifier when humidity >70%"
}

/**
```
```typescript
 * V11.2: Medication Adherence Pattern - Missed Dose vs. BP Rise
 * Example: "Missed BP medication → Blood pressure rises by 15 mmHg within 24h"
 */
interface MedicationAdherencePattern {
  userId: string;
  medicationName: string;
  medicationPurpose: 'blood_pressure' | 'diabetes' | 'heart_rate' | 'cholesterol';
  
  // Pattern
  trigger: {
    event: 'missed_dose' | 'late_dose' | 'double_dose';
    timeWindow: number; // Hours after trigger
  };
  
  // Health impact
  impact: {
    metric: 'blood_pressure' | 'blood_glucose' | 'heart_rate';
    change: number; // Absolute change (e.g., +15 mmHg)
    percentChange: number; // % change from baseline
  };
  
  // Evidence
  occurrences: number;
  confidence: number; // 0-100%
  
  // Recommendation
  recommendation: string;
}

/**
 * V11.2: Detect Medication Adherence Pattern
 */
function detectMedicationAdherencePattern(userId: string): MedicationAdherencePattern[] {
  const patterns: MedicationAdherencePattern[] = [];
  
  // Get 90 days of medication logs + health vitals
  const medLogs = getMedicationLogsLast90Days(userId);
  const vitals = getHealthVitalsLast90Days(userId);
  
  // Example: Missed BP medication → BP rise
  const missedDoses = medLogs.filter(log => log.status === 'missed' && log.medicationPurpose === 'blood_pressure');
  
  missedDoses.forEach(missedDose => {
    // Find BP readings 24h after missed dose
    const bpAfterMissed = vitals.filter(v => 
      v.bloodPressure && 
      v.timestamp >= missedDose.scheduledTime &&
      v.timestamp <= missedDose.scheduledTime + (24 * 60 * 60 * 1000)
    );
    
    // Calculate average BP change
    const baselineBP = getBaselineBloodPressure(userId);
    const avgSystolicChange = bpAfterMissed.reduce((sum, v) => sum + (v.bloodPressure!.systolic - baselineBP.systolic), 0) / bpAfterMissed.length;
    
    if (avgSystolicChange > 10 && bpAfterMissed.length >= 3) { // Significant rise, multiple occurrences
      patterns.push({
        userId,
        medicationName: missedDose.medicationName,
        medicationPurpose: 'blood_pressure',
        trigger: {
          event: 'missed_dose',
          timeWindow: 24, // 24 hours
        },
        impact: {
          metric: 'blood_pressure',
          change: avgSystolicChange,
          percentChange: (avgSystolicChange / baselineBP.systolic) * 100,
        },
        occurrences: bpAfterMissed.length,
        confidence: 85, // High confidence
        recommendation: 'Set multiple medication reminders. BP rises significantly when dose is missed.',
      });
    }
  });
  
  return patterns;
}
```

**Purpose:** Alert when vitals deviate >20% from personal baseline.

```typescript
/**
 * V11.2: Baseline Deviation Alert
 * Detect when current vital differs significantly from user's baseline
 */
interface BaselineDeviation {
  userId: string;
  metric: 'blood_pressure' | 'heart_rate' | 'blood_glucose' | 'weight';
  
  baseline: {
    value: number; // Average over last 30 days
    stdDev: number; // Standard deviation
    dataPoints: number; // Number of readings used
  };
  
  current: {
    value: number;
    timestamp: number;
  };
  
  deviation: {
    absolute: number; // Difference from baseline
    percentage: number; // % change
    zScore: number; // Number of standard deviations from mean
  };
  
  alert: {
    severity: 'minor' | 'moderate' | 'severe';
    notify: boolean;
    message: string;
  };
}

/**
 * V11.2: Detect Baseline Deviation
 */
function detectBaselineDeviation(
  userId: string,
  currentValue: number,
  metric: 'blood_pressure' | 'heart_rate' | 'blood_glucose' | 'weight'
): BaselineDeviation | null {
  // Calculate baseline (30-day average)
  const baseline = calculateBaseline(userId, metric, 30);
  
  if (!baseline || baseline.dataPoints < 7) {
    // Not enough data to establish baseline
    return null;
  }
  
  // Calculate deviation
  const absolute = currentValue - baseline.value;
  const percentage = (absolute / baseline.value) * 100;
  const zScore = absolute / baseline.stdDev;
  
  // Determine severity
  let severity: 'minor' | 'moderate' | 'severe' = 'minor';
  let notify = false;
  
  if (Math.abs(percentage) > 30 || Math.abs(zScore) > 3) {
    severity = 'severe'; // >30% or >3 standard deviations
    notify = true;
  } else if (Math.abs(percentage) > 20 || Math.abs(zScore) > 2) {
    severity = 'moderate'; // >20% or >2 standard deviations
    notify = true;
  }
  
  if (!notify) return null; // No significant deviation
  
  return {
    userId,
    metric,
    baseline: {
      value: baseline.value,
      stdDev: baseline.stdDev,
      dataPoints: baseline.dataPoints,
    },
    current: {
      value: currentValue,
      timestamp: Date.now(),
    },
    deviation: {
      absolute,
      percentage,
      zScore,
    },
    alert: {
      severity,
      notify,
      message: `⚠️ BASELINE DEVIATION - ${metric} is ${percentage.toFixed(0)}% ${absolute > 0 ? 'higher' : 'lower'} than baseline ` +
               `(${baseline.value.toFixed(1)}). Current: ${currentValue}. Monitor closely.`,
    },
  };
}
```

```typescript
/**
 * V11.2: Predictive Health Score
 * 0-100 score predicting likelihood of health event in next 7 days
 */
interface PredictiveHealthScore {
  userId: string;
  timestamp: number;
  score: number; // 0-100 (100 = no risk, 0 = imminent crisis)
  
  // Risk factors
  riskFactors: Array<{
    factor: string; // e.g., "Declining medication adherence"
    weight: number; // Contribution to risk (0-1)
    trend: 'worsening' | 'improving' | 'stable';
  }>;
  
  // Prediction
  prediction: {
    event: 'hypertensive_crisis' | 'hypoglycemia' | 'fall_risk' | 'medication_hospitalization';
    probability: number; // 0-1
    timeframe: number; // Days until predicted event
  };
  
  // Recommendation
  recommendation: string;
}

/**
 * V11.2: Calculate Predictive Health Score
 */
function calculatePredictiveHealthScore(userId: string): PredictiveHealthScore {
  let score = 100; // Start at perfect
  const riskFactors = [];
  
  // Factor 1: Medication adherence (weight: 0.3)
  const medAdherence = getMedicationAdherence7Days(userId);
  if (medAdherence < 90) {
    const penalty = (100 - medAdherence) * 0.3;
    score -= penalty;
    riskFactors.push({
      factor: 'Medication adherence declining',
      weight: penalty / 100,
      trend: 'worsening',
    });
  }
  
  // Factor 2: Vital trends (weight: 0.4)
  const bpTrend = getBloodPressureTrend7Days(userId);
  if (bpTrend.direction === 'increasing' && bpTrend.rate > 5) { // >5 mmHg/week
    const penalty = 40; // High weight for BP
    score -= penalty;
    riskFactors.push({
      factor: 'Blood pressure trending upward',
      weight: penalty / 100,
      trend: 'worsening',
    });
  }
  
  // Factor 3: Activity level (weight: 0.2)
  const activityLevel = getActivityLevel7Days(userId);
  if (activityLevel < 50) { // <50% of baseline
    const penalty = 20;
    score -= penalty;
    riskFactors.push({
      factor: 'Activity level decreased',
      weight: penalty / 100,
      trend: 'worsening',
    });
  }
  
  // Factor 4: Sleep quality (weight: 0.1)
  const sleepQuality = getSleepQuality7Days(userId);
  if (sleepQuality < 60) {
    const penalty = 10;
    score -= penalty;
    riskFactors.push({
      factor: 'Poor sleep quality',
      weight: penalty / 100,
      trend: 'worsening',
    });
  }
  
  score = Math.max(0, Math.min(100, score));
  
  // Predict event
  let prediction = null;
  if (score < 50 && bpTrend.direction === 'increasing') {
    prediction = {
      event: 'hypertensive_crisis' as const,
      probability: 0.7,
      timeframe: 7, // Within 7 days
    };
  }
  
  return {
    userId,
    timestamp: Date.now(),
    score,
    riskFactors,
    prediction: prediction || {
      event: 'medication_hospitalization' as const,
      probability: 0.1,
      timeframe: 30,
    },
    recommendation: score < 70 
      ? 'Multiple risk factors detected. Schedule doctor visit within 48 hours.'
      : 'Continue monitoring. Health status stable.',
  };
}
```
